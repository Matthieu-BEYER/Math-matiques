\section{Les suites arithmétiques}
\subsection{Définition par récurrence}

\begin{example}[Définition 1 :]
\noindent Une suite est dite \textbf{arithmétique} lorsqu'on passe d'un terme au suivant en ajoutant à chaque fois un même nombre $r$ appelé \textbf{raison} de la suite.\\

\noindent Une suite arithmétique est définie par la donnée de son premier terme (généralement $u_0$ ou $u_1$) et la relation de récurrence :\\
\[ \left\lbrace
   \begin{array}{l}
      u_0 = k \text{ avec k} \in \mathbb{R}\\
      u_{n+1} = u_n + r\\
   \end{array}
\right. \]
\end{example}

\begin{methode}[Exemple 1 :]
\noindent Une chaîne de supermarché avait 200 points de vente en 2010. Chaque année elle en a ouvert 6 de plus.\\

\noindent On note $u_n$ le nombre de points de vente de la chaîne au bout de n année. Le premier terme de la suite $(u_n)$ est $u_0=200$.

\noindent On passe d'un terme au suivant en ajoutant 6, c'est-à-dire : $u_{n+1}=u_n+6$.

\noindent $(u_n)$ est donc une suite arithmétique de raison 6.
\end{methode}

\begin{proposition}[Méthode 1 : montrer qu'une suite est arithmétique]
Pour démontrer qu'une suite est arithmétique, on calcule la différence entre deux termes consécutifs.\\
Si cette différence est constante, quelles que soient les valeurs de n, c'est a dire que sa valeur ne dépend pas de $n$, on peut en conclure que la suite $(u_n)$ est arithmétique.
\end{proposition}

\begin{methode}[Exemple 2 :]
Soit la suite $(w_n)$ définie par $w_{n}=3n+7$ pour tout entier naturel $n$.
\begin{eqnarray*}
w_{n+1}-w_n & = & 3\times (n+1) +7 - (3n+7) \quad = \quad 3n+3+7-3n-7 \quad = \quad 3
\end{eqnarray*}

\noindent Donc $(w_n)$ est une suite arithmétique de raison $r=3$
\end{methode}

\begin{propri}[Propriété 1 :]
Soit$(u_n)$ une suite arithmétique de raison $r$ et de premier terme $u_0$.
\begin{enumerate}
\item[-]Pour tout entier naturel$n$, on a :\\

$$u_n=u_0+n \times r$$ \\
\item[-]Pour tout entiers naturels $n$ et $p$, on a : \\

\vspace{-9mm}
$$u_n=u_p+(n-p) \times r$$
\end{enumerate}
\end{propri}


\begin{methode}[Exemples 3 :]
Soit $(u_n)$ la suite arithmétique définie sur $\N$ telle que $u_5=7$ et $r$ $=$ $3$\\
1. Calculer $u_9$. \\
 

$u_9=u_5+(9-5) \times r$, donc $u_9$ = $7$ $+$ $4$ $\times$ $3$ $=$ $19$ \\
\newline
Soit $(v_n)$ la suite arithmétique définie sur $\N$ telle que $v_1=10$ et $r$ $=$ $-2$\\
1. Exprimer $v_{n+1}$ en fonction de $v_n$. \\
$v_{n+1}$ = $v_n$ $+$ $r$ $=$ $v_n$ $-$ $2$. \\
\newline
2. Pour tout entier $n$ $\neq$  $0$, exprimer $v_n$ en fonction de $n$.  \\
$v_n$ $=$ $v_1$ $+$ $(n-1)r$ donc $v_n$ $=$ $10$ $-$ $2(n-1)$ \\
\newline
3. Donner la valeur de $v_9$. \\
$v_9$ $=$ $v_1$ $+$ $(9-1)r$ $=$ $10$ $-$ $2$ $\times$ $8$ $=$ $10$ $-$ $16$ $=$ $-6$

\end{methode}


\subsection{Propriétés des suites arithmétiques}

\subsubsection{Sens de variation}

\begin{propri}[Propriété 2 :]
Soit $(u_n)$ une suite arithmétique de raison $r$ alors :

\begin{enumerate}
\item[-] Si $r>0$ alors $(u_n)$ est \textbf{strictement croissante.}
\item[-] Si $r=0$ alors $(u_n)$ est \textbf{constante.}
\item[-] Si $r<0$ alors $(u_n)$ est \textbf{strictement décroissante.}
\end{enumerate}
\end{propri}

\noindent \textbf{Démonstration :}
On considère une suite arithmétique de raison $r$. On a donc pour tout $n \in \N$ :
$$u_{n+1}=u_n+r \Longleftrightarrow u_{n+1}-u_n = r$$
\noindent On observe trois cas possible :
\begin{enumerate}
\item[-] Si $r>0$ alors pour tout $n\in \N$, $u_{n+1}-u_n >0$ donc $(u_n)$ est strictement croissante.
\item[-] Si $r=0$ alors pour tout $n\in \N$, $u_{n+1}-u_n =0$ donc $(u_n)$ est constante.
\item[-] Si $r<0$ alors pour tout $n\in \N$, $u_{n+1}-u_n <0$ donc $(u_n)$ est strictement décroissante.
\end{enumerate}


\begin{methode}[Exemple 4 :]
Soit la suite $(w_n)$ définie par $w_{n}=3n+7$ pour tout entier naturel $n$. 

\noindent On a vu ci-dessus que $(w_n)$ est une suite arithmétique de raison $r=3$.

\noindent Donc $(w_n)$ est une suite strictement croissante car $r>0$

\medskip
\noindent On peut retrouver ce résultat grâce à la définition du sens de variation :
\begin{eqnarray*}
w_{n+1}-w_n & = & 3\times (n+1) +7 - (3n+7)\\
&=&3n+3+7-3n-7\\
&=&3>0
\end{eqnarray*}

\noindent Donc $(w_n)$ est une suite arithmétique croissante.
\end{methode}

\subsection{Lien avec la moyenne arithmétique}

\begin{example}[Définition 2:]
\noinent La moyenne arithmétique de deux réels $a$ et $b$ est égale à $\dfrac{a+b}{2}$
\end{example}

\begin{propri}[Propriété 4 :]
\noindent Pour que trois réels $x$, $y$ et $z$ soient les termes consécutifs d'une suite arithmétique, il faut et il suffit que le réel $y$ soit égal à la moyenne arithmétique des réels $x$ et $z$.
\end{propri}

\begin{methode}[Exemple 6 :]
\noindent Puisque $\dfrac{1 + 9}{2}$ $=$ $5$, alors $1$, $5$ et $9$ sont les termes consécutifs d'une suite arithmétique.
\end{methode}

\subsection{Somme de termes consécutifs d'une suite arithmétique}


\begin{propri}[Propriété 5 :]
\noindent La somme $S$ des $n$ termes consécutifs d'une suite arithmétique est égale à : \\
\begin{center}
$S$ = $n$ $\times$ $\dfrac{premier~terme~+~dernier~terme}{2}$
\end{center}
\end{propri}

\begin{proposition}[Méthode 2 : calculer la somme d'une suite arithmétique]
\noindent On considère pour tout entier naturel $n$ la suite arithmétique $(u_n)$. \\
$S$ = $\textcolor{blue}{u_0}$ $+$ $u_1$ $+$ $u_2$ $+$ ... $+$ $\textcolor{red}{u_n}$ $=$ $\sum_{k=0}^{n}u_k$ = $(n+1)$ $\times$ $\dfrac{\textcolor{blue}{u_0}+\textcolor{red}{u_n}}{2}$
\end{proposition}

\end{propri}


\begin{methode}[Exemple 7 :]
\noindent Soit $(u_n)$ la suite arithmétique de premier terme $u_0$ $=$ $19$ et de raison $r$ $=$ $5$. \\
1. Calculer $S_{10}$ $=$ $\sum_{k=0}^{10}u_k$. \\
\newline
Le premier terme est $u_0$ $=$ $19$. \\
La somme comporte $11$ termes. \\
Le dernier terme est $u_{10}$ $=$ $u_0$ $+$ $10$ $\times$ $5$ $=$ $69$. \\
Donc $S_{10}$ $=$ $11$ $\times$ $\dfrac{u_0 + u_{10}}{2}$ $=$ $11$ $\times$ $\dfrac{19 + 69}{2}$ $=$ $484$
\end{methode}







\section{Suites géométriques}

\subsection{Définition}

\begin{example}[Définition 3 :]
Une suite $\suiten$ est dite \textbf{géométrique} si l'on passe d'un terme au suivant en \textbf{multipliant toujours par le même nombre} $q$, appelé ici aussi \textbf{raison} de la suite.
\noindent Autrement dit, une suite est \textbf{géométrique} si sa formule de récurrence est du type : \[u_{n+1}=u_n \times q \quad \text{ pour tout } n \geq 0.\]
\end{example}

\begin{methode}[Exemple 8 :]
La suite des puissances de dix : $1-10-100-1000-10000-\ldots$ est géométrique de raison 10.\\
\end{methode}

\noindent \textbf{Remarque :} \\
Une suite pour laquelle on passe d'un terme au suivant en divisant toujours par le même nombre est également géométrique, puisque diviser par un nombre revient à multiplier par son inverse.


\begin{proposition}[Méthode 3 :]
Pour démontrer qu'une suite est géométrique :\\
\begin{itemize}
	\item soit on transcrit le texte de l'énoncé sous forme de formule de récurrence ;
	\item soit on calcule le \textbf{quotient} $\dfrac{u_{n+1}}{u_n}$ entre deux termes consécutifs et on montre que ce quotient est constant et ne dépend pas de $n$.\\
	
\noindent Le résultat de ce calcul est alors la \textbf{raison} $q$ de la suite.
\end{itemize}
\end{proposition}

\begin{methode}[Exemple 9 :]
On souhaite démontrer que la suite définie par $u_n=5\times 3^n$ pour tout $n \pg 0$ est géométrique.\\
\begin{itemize}
	\item on calcule $u_{n+1}$ : $u_{n+1}=5\times 3^{n+1}=5 \times 3^n \times 3$ ;
	\item puis le quotient $\frac{u_{n+1}}{u_n}=\frac{5 \times 3^n \times 3}{5\times 3^{n}}=3$.
\end{itemize}
Le résultat ne dépend plus de $n$, donc la suite est \textbf{géométrique} de raison $q=3$.
\end{methode}

\subsection{Formule explicite}

\begin{propri}[Propriété 5 :]
Soit $\suiten$ une suite \textbf{géométrique} de raison $q$ et de premier terme $u_0$.

Alors on a :\[u_n=u_0\times q^n \quad  \text{ pour tout } n \in \mathbb{N}.\]
\end{propri}

\noindent \textbf{Remarque :}\\
Si le premier terme de la suite n'est pas $u_0$ mais $u_1$, on a la propriété : \[u_n=u_1 \times q^{n-1} \quad \text{ pour tout }n \geq 1\]%
et, pour n'importe quel rang $p$ : \[u_n=u_p \times q^{n-p} \quad \text{ pour tout }n \geq p.\]

\newpage

\begin{methode}[Exemple 10 :]
Je place en banque une somme de 350 \euro{}. Au 1\ier janvier de chaque année, je perçois sur mon compte des intérêts qui s'élèvent à 3\,\% de la somme placée pendant l'année écoulée.  Quelle somme sera disponible sur mon compte au bout de 5 ans ?
\begin{itemize}
	\item Au bout d'un an, il y aura $350+\frac{3}{100}\times 350=350+10,5=360,5$ \euro{}.
	\item Au bout de deux ans, il y aura $306,5+\frac{3}{100}\times 360,5=360,5+10,815 \approx 371,31 $ \euro{}.
	\item La suite $(c_n)]$ où $c_n$ est la somme placée à la fin de la $n$-ième année est une suite géométrique de raison $q=1,03$ et de premier terme $c_0=350$.
	
	En effet, $c_{n+1}=c_n+\frac{3}{100} \times c_n=c_n+0,03c_n=1,03c_n$. 
	\item Au bout de 5 ans, il y aura donc sur le compte $c_{5}=c_0 \times q^ 5 = 350 \times 1,03^ 5 \approx 405,75 $ \euro{}.
\end{itemize}
\end{methode}

\subsection{Sens de variation}

\begin{propri}[Propriété 6 :]
\begin{itemize}
	\item Si $0$ $<$ $q$ $<$ $1$ alors ($u_n$) est strictement décroissante.
	\item Si $q$ $=$ $1$ alors ($u_n$) est constante.
	\item Si $q$ $>$ $1$ alors ($u_n$) est strictement croissante
\end{itemize}
\end{propri}
